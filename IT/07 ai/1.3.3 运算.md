# 算术运算

```
# 逻辑运算符号<、 >、|、 &
# 例如筛选p_change > 2的日期数据
data[data["p_change"] > 2].head()

# 完成一个多个逻辑判断， 筛选p_change > 2并且low > 15
data[(data["p_change"] > 2) & (data["low"] > 15)].head()
data.query("p_change > 2 & low > 15").head()

# 筛选'turnover'值为4.19, 2.39的行
data[data["turnover"].isin([4.19, 2.39])]
```
