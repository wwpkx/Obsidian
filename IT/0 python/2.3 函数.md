# 函数定义
```
# pass 语句不执行任何操作
def my_function():
  pass

def my_function(country = "China"):
  print("I am from " + country)

# *参数, 是一个元组
def func(*args):
	print(args)

**参数 是一个字典
def func(**args):
	print(args)
```

# lambda 函数
```
lambda _arguments_ : _expression_

x = lambda a : a + 10
print(x(5))

x = lambda a, b : a * b
print(x(5, 6))

sum = lambda arg1, arg2: arg1 + arg2 
print(sum(10, 20))
```
